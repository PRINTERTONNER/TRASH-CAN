import numpy as np

t = np.array([0.,1.,2.,3.,4.,5.,6.])
            # 1  2  3  4  5  6  7

print(t.ndim)  #rank를 표시 ------------------ 1로 나옴
print(t.shape)  #shpe를 표시 --------------- [7,]로 나옴
print(t[0], t[1], t[-1])  #0.0   ,   1.0   ,   6.0
print(t[2:5], t[4:-1])  #[ 2.   3.   4.] ----------- [ 4.   5.]
print(t[:2], t[3:])   # [0.    1.] ------------ [ 3. 4. 5. 6.]
